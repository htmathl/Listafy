podemos tentar incluir esse código no delete-todo.php

<?php
$id = $_GET['id']; // Obtém o ID do to-do a partir da URL

$con = mysqli_connect("localhost", "username", "password", "database"); // Substitua por suas credenciais de banco de dados

if (!$con) {
    die('Não foi possível conectar: ' . mysqli_error());
}

$sql = "DELETE FROM todo WHERE id_nota = $id"; // Substitua 'id_nota' pelo nome da coluna de ID no seu banco de dados

if (mysqli_query($con, $sql)) {
    echo "To-do deletado com sucesso";
} else {
    echo "Erro ao deletar to-do: " . mysqli_error($con);
}

mysqli_close($con);
?>
